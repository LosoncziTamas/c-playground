int Max(int x, int y)
{
    return x > y ? x : y;
}

PrintSubString(const char* str, int start, int end)
{
    assert(start <= end);
    for (char c = 0; start <= end; ++start)
    {
        c = *(str + start);
        putchar(c);
    }
    if (*(str + start) != '\n')
    {
        putchar('\n');
    }
}
